## 📊 Database Schema

Create a Prisma schema with a `Waitlist` model:

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model Waitlist {
  id        String   @id @default(uuid())
  firstName String
  lastName  String
  email     String   @unique
  role      String
  message   String
  status    String   @default("Waitlist")
  createdAt DateTime @default(now())
}
```

## 🔧 Backend Implementation

### **server.js**

```javascript
import express from "express";
import cors from "cors";
import helmet from "helmet";
import dotenv from "dotenv";
import waitlistRoutes from "./routes/waitlist.js";

dotenv.config();

const app = express();
app.use(express.json());
app.use(cors());
app.use(helmet());

app.use("/api/waitlist", waitlistRoutes);

app.get("/api/health", (req, res) => {
  res.json({ status: "Xenios API is running" });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
```

### **routes/waitlist.js**

```javascript
import express from "express";
import { PrismaClient } from "@prisma/client";
import { z } from "zod";
import sendConfirmationEmail from "../services/email.js";

const router = express.Router();
const prisma = new PrismaClient();

const waitlistSchema = z.object({
  firstName: z.string().min(1),
  lastName: z.string().min(1),
  email: z.string().email(),
  role: z.string().min(1),
  message: z.string().min(1),
});

router.post("/", async (req, res) => {
  try {
    const data = waitlistSchema.parse(req.body);

    const existing = await prisma.waitlist.findUnique({
      where: { email: data.email },
    });

    if (existing) {
      return res.status(400).json({
        error: "You are already on the waitlist.",
      });
    }

    const entry = await prisma.waitlist.create({
      data,
    });

    await sendConfirmationEmail(data);

    res.status(200).json({
      success: true,
      message: "Successfully joined the waitlist!",
      entry,
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: "Server error" });
  }
});

router.get("/", async (req, res) => {
  const entries = await prisma.waitlist.findMany({
    orderBy: { createdAt: "desc" },
  });
  res.json(entries);
});

export default router;
```

## 🌐 Frontend Integration

Add this script to your website:

```html
<script>
document.querySelector("form").addEventListener("submit", async (e) => {
  e.preventDefault();

  const formData = {
    firstName: document.querySelector('input[placeholder="First name"]').value,
    lastName: document.querySelector('input[placeholder="Last name"]').value,
    email: document.querySelector('input[type="email"]').value,
    role: document.querySelector("select").value,
    message: document.querySelector("textarea").value,
  };

  const response = await fetch("/api/waitlist", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
    },
    body: JSON.stringify(formData),
  });

  const result = await response.json();
  alert(result.message || "Submission successful!");
});
</script>
```
